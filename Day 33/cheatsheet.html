🌐 requests – HTTP GET

import requests

# Základný GET
response = requests.get("https://api.example.com/data")
response.raise_for_status()   # vyhodí výnimku pri 4xx/5xx
data = response.json()        # dict z JSON odpovede

# GET s parametrami (query string)
params = {"lat": 48.15, "lng": 17.14}
response = requests.get(url, params=params)
# → url?lat=48.15&lng=17.14
raise_for_status() – vždy volaj hneď po .get(), aby si hneď videl chybu namiesto tichého zlyhania.

📧 smtplib – Posielanie emailov

import smtplib

with smtplib.SMTP("smtp.gmail.com") as conn:
    conn.starttls()           # šifrovaný kanál
    conn.login(user=MY_EMAIL,
               password=PASSWORD)
    msg = (f"Subject:Look up!\n\n"
           "The ISS is above you!"
          ).encode("utf-8")
    conn.sendmail(
        from_addr=MY_EMAIL,
        to_addrs=MY_EMAIL,
        msg=msg
    )
Formát správy: Subject:Predmet\n\nTelo – dvojitý newline oddeľuje hlavičku od tela.

🖼️ tkinter Canvas + API tlačidlo

from tkinter import *
import requests

def get_quote():
    r = requests.get("https://api.kanye.rest/")
    quote = r.json()["quote"]
    canvas.itemconfig(quote_text, text=quote)

window = Tk()
canvas = Canvas(width=300, height=414)
bg_img = PhotoImage(file="background.png")
canvas.create_image(150, 207, image=bg_img)
quote_text = canvas.create_text(
    150, 207, text="...", width=250,
    font=("Arial", 30, "bold"), fill="white"
)
canvas.grid(row=0, column=0)

btn_img = PhotoImage(file="kanye.png")
btn = Button(image=btn_img, command=get_quote,
             highlightthickness=0)
btn.grid(row=1, column=0)

window.mainloop()
