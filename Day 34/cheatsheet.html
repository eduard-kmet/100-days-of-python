🧠 QuizBrain – Herná logika

class QuizBrain:

    def __init__(self, q_list):
        self.question_number = 0
        self.score = 0
        self.question_list = q_list
        self.current_question = None

    def still_has_questions(self):
        return self.question_number < len(self.question_list)

    def next_question(self):
        self.current_question = \
            self.question_list[self.question_number]
        self.question_number += 1
        q_text = html.unescape(
            self.current_question.text
        )
        return f"Q.{self.question_number}: {q_text}"

    def check_answer(self, user_answer):
        correct = self.current_question.answer
        if user_answer.lower() == correct.lower():
            self.score += 1
            return True
        return False

🔡 html.unescape()

import html

# API vracia HTML entity v texte:
raw = "Which country &amp; city hosted 2016?"
clean = html.unescape(raw)
# → "Which country & city hosted 2016?"

# Ďalšie príklady entity:
# &lt;   →  <
# &gt;   →  >
# &quot; →  "
# &#039; →  '
Prečo? Open Trivia DB (a väčšina API) posiela text s HTML escape sekvenciami. html.unescape() ich prevedie na normálne znaky.

⏱️ window.after() – Oneskorená akcia

# Zavolaj funkciu po N milisekundách
self.window.after(1000, self.get_next_question)
# → po 1 sekunde zavolá get_next_question()

# Prečo nie time.sleep()?
# sleep() ZAMRAZÍ celé okno (neodpovedá)
# after() je neblokujúce – okno zostáva živé

def give_feedback(self, is_right):
    color = "green" if is_right else "red"
    self.canvas.config(bg=color)
    self.window.after(1000,
                       self.get_next_question)

🖼️ QuizInterface – Tkinter UI

class QuizInterface:

    def __init__(self, quiz_brain: QuizBrain):
        self.quiz = quiz_brain
        self.window = Tk()
        self.window.config(
            padx=20, pady=20, bg=THEME_COLOR
        )
        # Label pre skóre
        self.score_label = Label(
            text="Score: 0",
            fg="white", bg=THEME_COLOR
        )
        self.score_label.grid(row=0, column=1)
        # Canvas s textom otázky
        self.canvas = Canvas(
            width=300, height=250, bg="white"
        )
        self.question_text = self.canvas.create_text(
            150, 125, width=200,
            font=("Arial", 10, "italic"),
            fill=THEME_COLOR
        )
        self.canvas.grid(
            row=1, column=0, columnspan=2
        )
