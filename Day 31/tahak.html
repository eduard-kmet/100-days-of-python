# Dve nesuvisiace triedy — NEDEDIA od seba!
class Polozka:
    def __init__(self, nazov, cena):
        self.nazov = nazov
        self.cena = cena

class Spravca:
    def __init__(self):
        self.zoznam = []          # interni zoznam objektov

    def pridaj(self, nazov, cena):
        self.zoznam.append(Polozka(nazov, cena))  # vytvor objekt!

    def vypis(self):
        for p in self.zoznam:
            print(p.nazov, p.cena)   # pristup cez atributy

Pandas

import pandas as pd

# Nacitanie / ulozenie
df = pd.read_csv("subor.csv")
df.to_csv("novy.csv", index=False)

# Pristup k datam
df["meno"]                        # jeden stlpec
df[df["oddelenie"] == "IT"]       # filtrovanie riadkov

# Filtrovanie — pozor: & nie and, zatvorky povinne!
df[(df["oddelenie"] == "IT") & (df["plat"] > 2700)]

# Zakladne operacie
df["plat"].mean()    # priemer
df["plat"].max()     # maximum
df["plat"].min()     # minimum
df["plat"].sum()     # sucet

# Groupby
df.groupby("oddelenie")["plat"].mean()    # priemer platu podla oddelenia
df.groupby("oddelenie")["meno"].count()   # pocet v oddeleni

# Zamestnanec s max hodnotou v kazdom oddeleni
df.loc[df.groupby("oddelenie")["roky"].idxmax()]

# Novy stlpec s podmienkou (lambda)
df["bonus"] = df.apply(lambda row: row["plat"] * 0.05 if row["roky"] > 3 else row["plat"] * 0.02, axis=1)
